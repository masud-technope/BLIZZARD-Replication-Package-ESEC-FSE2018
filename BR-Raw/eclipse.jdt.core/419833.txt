@Target(ElementType.TYPE_USE)
@Retention(RetentionPolicy.RUNTIME)
@Documented
public @interface TypeUse {

}
public class Shape {
}
public class Square extends @TypeUse(1) Shape implements java.awt.@TypeUse Shape{
}
