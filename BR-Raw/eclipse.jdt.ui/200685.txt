package test;
public class E01 {
public void foo() {
"A foo bar text"
}
}

Add a template 'Test' with pattern:
foo(${word_selection});
bar();

package test;

package test;
public class E01 {
public void foo() {
foo("A foo bar text");
bar();
}
}
